package org.example;

import static java.lang.Math.abs;

public class Chess {
    public static int kingX;
    public static int kingY;
    public static int rookX;
    public static int rookY;
    public static int bishopX;
    public static int bishopY;
    public static String result;

    public Chess(int kingX, int kingY, int rookX, int rookY, int bishopX, int bishopY, String result) {
        this.kingX = kingX;
        this.kingY = kingY;
        this.rookX = rookX;
        this.rookY = rookY;
        this.bishopX = bishopX;
        this.bishopY = bishopY;
        this.result = result;
    }

    public static  String solve(int kingX, int kingY, int rookX, int rookY, int bishopX, int bishopY) {

        //     РџСЂРѕРІРµСЂСЏРµРј, СѓРіСЂРѕР¶Р°РµС‚ Р»Рё РєРѕСЂРѕР»СЋ Р»Р°РґСЊСЏ РёР»Рё СЃР»РѕРЅ
        boolean threatenedByRook = (kingX == rookX) || (kingY == rookY);
        boolean threatenedByBishop = (abs(kingX - bishopX) == abs(kingY - bishopY));

        // Р’С‹РІРѕРґРёРј СЂРµР·СѓР»СЊС‚Р°С‚
        if (threatenedByRook && threatenedByBishop) {
            return "both";
        } else if (threatenedByRook) {
            return "rook";
        } else if (threatenedByBishop) {
            return "bishop";
        } else {
            return "none";
        }
    }

    @Override
    public String toString(){
        return kingX + " " + kingX + " " + rookX + " " + rookY + " " + bishopX + " " + bishopY + " " + result;
    }
}
